```javascript
app.post('/admin_api/server/restart', async (req, res) => {
    // 1. 立即返 202 (shutdown 异步, 不能 hang 客户端)
    res.status(202).json({
        status: 'accepted',
        message: '主服务正在执行优雅重启。v-services-restart watchdog 会在进程退出后自动拉起。',
        reason: 'admin_restart',
        timestamp: new Date().toISOString()
    });
    // 2. setImmediate 让响应先 flush, 再触发 graceful shutdown
    setImmediate(() => {
        console.log(`[Server] Restart triggered from admin API. exit_code=1, reason=admin_restart`);
        gracefulShutdown(1, 'admin_restart').catch(err => {
            console.error('[Server] Fatal error during graceful restart:', err);
            process.exit(1);
        });
    });
});
```
